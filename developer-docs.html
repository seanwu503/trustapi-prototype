const client = new TrustAPI({

Verify User Endpoint

The primary endpoint to check if a user meets your trust requirements.

Request

POST /v1/verify 
const result = await client.verify({
  wallet: "0x742d35Cc6634C0532925a3b844Bc9e7595f0bEb7",
  minimumTier: "silver",
  checks: ["human", "trust"]
});

// Result
{
  verified: true,
  humanLikelihood: "high",
  trustTier: "gold",
  confidence: 0.92,
  proofExpires: "2026-03-04T12:00:00Z"
}

Parameters

Parameter Type Required Description
wallet string Required The wallet address or user identifier to verify
minimumTier enum Optional Minimum trust tier: "bronze", "silver", or "gold"
checks array Optional Specific checks to run: ["human", "trust", "age"]

Response

✓ Successful Response
{
  "verified": true,
  "humanLikelihood": "high",
  "trustTier": "gold",
  "confidence": 0.92,
  "accountAge": 847,
  "activityScore": 8.7,
  "proofId": "0xA7F9...3C2E",
  "proofExpires": "2026-03-04T12:00:00Z"
}
⚠️ Important Note
Proofs are time-bound and expire after 30 days. Users can revoke their proofs at any time. Always check the proofExpires field and implement webhook listeners for revocations.

SDK Examples

React Integration

React 
import { useTrustAPI } from '@trustapi/react';

function LoginButton() {
  const { verify, loading, verified } = useTrustAPI();

  const handleLogin = async () => {
    const result = await verify({
      minimumTier: 'silver'
    });

    if (result.verified) {
      // Proceed with login
      handleSuccessfulAuth(result);
    }
  };

  return (
    <button onClick={handleLogin} disabled={loading}>
      {verified ? 'Verified ✓' : 'Verify Identity'}
    </button>
  );
}

Node.js Backend

Node.js 
const { TrustAPI } = require('@trustapi/sdk');

const client = new TrustAPI({
  apiKey: process.env.TRUST_API_KEY
});

app.post('/api/signup', async (req, res) => {
  const { wallet } = req.body;

  try {
    const result = await client.verify({
      wallet,
      minimumTier: 'silver',
      checks: ['human', 'trust']
    });

    if (!result.verified) {
      return res.status(403).json({
        error: 'User does not meet trust requirements'
      });
    }

    // Create user account
    const user = await createUser({ wallet, trustTier: result.trustTier });
    res.json({ success: true, user });

  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
